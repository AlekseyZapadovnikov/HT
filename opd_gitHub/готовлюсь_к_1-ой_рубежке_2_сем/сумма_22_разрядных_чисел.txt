ORG 0x0030

SUM22:
	LD $COUNTER1
	ADD $COUNTER3
	ST RES1
	BCS CARRY_1
SUDA:
	LD $COUNTER2
	ADD $COUNTER4
	ADD $RES2
	AND $MASK
	ST $RES2
	HLT

COUNTER1: WORD 0xAAAB
COUNTER2: WORD 0x001C
COUNTER3: WORD 0xA316
COUNTER4: WORD 0x002A

RES1: WORD 0x0000
RES2: WORD 0x0000
MASK: WORD 0x003F

CARRY_1:
	LD RES2
	INC
	ST RES2
	jump SUDA
