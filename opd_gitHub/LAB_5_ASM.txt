ORG 0x3ED
START: CLA

S1:
	IN 3
	AND #0x40
	BEQ S1
	LD (ITERATOR)+
	SWAB
	OUT 2
	SWAB
	ST CHAR
	CALL S2

S2:
	AND #0x0F
	SUB #0x0A
	BEQ EXIT
	IN 3
	AND #0x40
	BEQ S2
	LD CHAR
	OUT 2
	CALL S1

EXIT:
	HLT

CHAR: WORD 0x0000
ITERATOR: WORD 0x059F

ORG 0x59F
CHARS:
	OT: WORD 0xDEE2
	CHA: WORD 0xE7D0
	YAN: WORD 0xEFDD
	NI: WORD 0xDDEB
	UWU: WORD 0xD90A
